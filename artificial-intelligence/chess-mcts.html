def run_mcts_flow(current_board, time_limit):
root_node = create_starting_circle(current_board)
while timer_is_still_running(time_limit):
current_node = root_node
# --- PHASE 1: SELECTION ---
while all_moves_tested(current_node):
current_node = apply_ucb1_and_go_down(current_node)
if game_is_over(current_node): break
# --- PHASE 2: EXPANSION ---
if not game_is_over(current_node):
leaf_node = pick_untested_move(current_node)
else: leaf_node = current_node
# --- PHASE 3: SIMULATION ---
game_result = play_random_game(leaf_node)
# --- PHASE 4: BACKPROPAGATION ---
climb_up_and_add_result(leaf_node, game_result)
# --- FINAL CHOICE ---
best_move = highest_visit_child(root_node)
return best_move
